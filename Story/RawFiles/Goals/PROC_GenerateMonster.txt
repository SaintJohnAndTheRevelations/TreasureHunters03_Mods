PROC
PROC_GenerateMonster((GUIDSTRING)_spawnPoint)
AND
SysCount("DB_MonstersToFight", 2, (INTEGER)_totalFightMonsters)
AND
Random(_totalFightMonsters, _modulo)
AND
Random(_modulo, _monsterIndex)
AND
DB_MonstersToFight(_monsterIndex, _monsterToFight)
AND
DB_IsPlayer(_character)
AND
CharacterGetLevel(_character, _characterLevel)
THEN
CharacterAppearAt((CHARACTERGUID)_monsterToFight, _spawnPoint, 1, ""); 
PlayEffect(_monsterToFight, "RS3_FX_Char_Ghosts_Teleport_in_01");
SetFaction((CHARACTERGUID)_monsterToFight, "Good NPC");
CharacterLevelUpTo((CHARACTERGUID)_monsterToFight, _characterLevel);
CharacterFreeze((CHARACTERGUID)_monsterToFight);
DB_FrozenMonsters((CHARACTERGUID)_monsterToFight);
TimerLaunch("MONSTERFREEZE", 4000);
TimerLaunch("UNFREEZECHARACTERS", 16000);
